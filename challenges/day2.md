## BASIC FOR LOOPS & CONDITIONALS

0. all_blue

  start:  ['.', '.', '.', '.', '.']  
  finish: ['b', 'b', 'b', 'b', 'b']  

1. all_first_color

  start:  ['b', '.', '.', '.', '.']  
  finish: ['b', 'b', 'b', 'b', 'b']  

  start:  ['g', '.', '.', '.', '.']  
  finish: ['g', 'g', 'g', 'g', 'g']  

2. n_in_a_row

  start:  ['.', '.', '.', '.', '.']  
  finish: ['b', 'b', '.', '.', '.']  \(given n is 2)  
  finish: ['b', 'b', 'b', '.', '.']  \(given n is 3)  

3. every_even_erase

  start:  ['b', 'b', 'b', 'b', 'b']  
    finish: ['.', 'b', '.', 'b', '.']  
  
4. every_odd_erase

  start:  ['b', 'b', 'b', 'b', 'b']  
    finish: ['b', '.', 'b', '.', 'b']  
  
5. every_n_erase

  start:  ['b', 'b', 'b', 'b', 'b']  
    finish: ['b', 'b', 'b', '.', 'b']  \(given n is 4)  
  finish: ['b', 'b', 'b', 'b', '.']  \(given n is 5)  

6. move_the_blue_dot_one_to_the_right

  start:  ['.', '.', 'b', '.', '.']  
    finish: ['.', '.', '.', 'b', '.']  
  
  start:  ['g', 'b', '.', '.', 'g']  
    finish: ['g', '.', 'b', '.', 'g']  
  
7. delete_before

  start:  [ 'g', 'b', '.', '.', '.' ]    
  finish: [ '.', 'b', '.', '.', '.' ]    

  start:  [ 'g', 'b', 'b', '.', '.' ]    
  finish: [ '.', '.', 'b', '.', '.' ]
